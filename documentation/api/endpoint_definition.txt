Endpoint: /auth/
    Register:
        POST /auth/register/
        | JSON
        | - Benutzername: String
        | - Password: String
        Response:
        | 201
        | JSON
        | - Auth-Token: String

    !Login:
        POST /auth/login/
        | JSON
        | - Benutzername: String
        | - Password: String
        Response:
        | 200
        | JSON
        | - Auth-Token: String # Permanently saved


    !Token:
        GET /auth/token/
        | HEADER
        | - Auth-Token: String
        Response:
        | 200
        | JSON:
        | - Backend-Token: String
        | - Validation Time: String

Endpoint: /fridge/
    !Get all Fridges:
        GET /fridge/
        | HEADER
        | - Backend-Token: String
        Response:
        | 200
        | JSON 
        | - List of all Fridges: List of Objects
        |       - Name: String
        |       - Description: String
        |       - Item Data: Object
        |           - Expire Date: Date
    
Endpoint: /fridge/management/
    Join Fridge:
        POST /fridge/management/join/
        | HEADER
        | - Backend-Token: String
        | JSON
        | - Fridge-Id: String
        | Response:
        | 201
        | "Joined"

    !Create Fridge:
        POST /fridge/management/create/
        | HEADER
        | - Backend-Token: String
        | JSON
        | - Fridge Name: String
        | - Description: String
        | Response
        | 201
        | "Created"        

    Edit Fridge:
        POST /fridge/management/
        | HEADER
        | - Backend-Token: String
        | JSON
        | - Fridge-Id: String
        | - Name: String
        | - Description: String
        | Response
        | 200
        | JSON
        | - Fridge-Name: String
        | - Description: String

    Remove Fridge:
        DELETE /fridge/management/<fridge-id>
        | HEADER
        | - Backend-Token: String
        | # Leaves Fridge, if Fridge group is empty delete fridge
        | Reponse:
        | 200
        | "Left Successful"

    !Get Content:

        | - Store: String
